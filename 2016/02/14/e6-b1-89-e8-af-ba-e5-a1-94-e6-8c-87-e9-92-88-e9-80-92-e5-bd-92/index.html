#include <iostream>

#include<cstdio>

#include<cstring>
using namespace std;
int t,n;
int a[70],b[70],c[70];
int m,p,q;
int flag;
int main()
{
void fuc(int a,int b,int *c,int n);

scanf(“%d”,&t);
while(t–)
{
scanf(“%d”,&n);
scanf(“%d”,&a[0]);
for(int i=1;i<=a[0];i++)
scanf(“%d”,&a[i]);
scanf(“%d”,&b[0]);
for(int i=1;i<=b[0];i++)
scanf(“%d”,&b[i]);
scanf(“%d”,&c[0]);
for(int i=1;i<=c[0];i++)
scanf(“%d”,&c[i]);

fuc(a,b,c,n);
if( flag )
puts(“true”);
else
puts(“false”);

}
return 0;
}

void fuc(int a,int b,int *c,int n)
{

if(n==0)
{
flag = 1;
return ;
}

if(b[1]==n && b[0])
{
flag = 0;
return ;
}

if(a[0] && n==a[1]) //最大的盘子还没有挪好

{

a[1] = a[0] - 1; //去掉最大的盘子，再进行判断；

fuc(++a,c,b,n-1);

}else if(c[0] && n==c[1]) //最大的已经挪好了

{

c[1] = c[0] - 1; //去掉最大的盘子再进行判断

fuc(b,a,++c,n-1); //++c代表之后的c[0变成现在的c[1]；

}

}


}
